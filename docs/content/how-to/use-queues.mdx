## Goal

Decouple producers from consumers using the Queue module. Publish messages to a topic, and let queue-triggered functions process them asynchronously.

## Steps

### 1. Enable the Queue module

{/* TODO: Move all config details to their own file and have more minimal examples on the HOWTOs */}

```yaml title="iii-config.yaml"
modules:
  - class: modules::queue::QueueModule
    config:
      adapter:
        class: modules::queue::BuiltinQueueAdapter
        config:
          store_method: file_based  # Options: in_memory, file_based
          file_path: ./data/queue_store  # required for file_based
          max_attempts: 3          # Max retries before sending to DLQ (default: 3)
          backoff_ms: 1000         # Base backoff delay in ms (default: 1000)
          concurrency: 10          # Global concurrent workers (default: 10)
          poll_interval_ms: 100    # How often to poll for jobs in ms (default: 100)
          mode: "concurrent"       # "fifo" or "concurrent" (default: "concurrent")

        # Redis adapter

        # config:
        #   redis_url: redis://localhost:6379
        #
        # RabbitMQ adapter

        #       amqp_url: "amqp://localhost:5672"   # RabbitMQ connection URL

        #       prefetch_count: 10                   # QoS prefetch count (default: 10, forced to 1 in FIFO mode)

```

```typescript title="queue-producer.ts"
import { init, getContext } from 'iii-sdk'

const iii = init(process.env.III_BRIDGE_URL ?? 'ws://localhost:49134')

iii.registerFunction({ id: 'orders::create' }, async (input) => {
  const { logger } = getContext()
  const order = { id: crypto.randomUUID(), ...input }

  iii.triggerVoid('enqueue', {
    topic: 'order.created',
    data: order,
  })

  logger.info('Order enqueued', { orderId: order.id })
  return { orderId: order.id }
})

// Then call from another function or worker
const { orderId } = await iii.trigger('orders::create', { items: ['item-1', 'item-2'], total: 99.99 })
logger.info('Created order', { orderId })
```

<Tab value="Python">
```python title="queue_producer.py"
import asyncio
import os
import signal
import uuid
from iii import III
from iii.context import get_context

iii = III(os.environ.get('III_BRIDGE_URL', 'ws://localhost:49134'))

async def create_order(input):
    ctx = get_context()
    order = {"id": str(uuid.uuid4()), **input}

    iii.trigger_void("enqueue", {
        "topic": "order.created",
        "data": order,
    })

    ctx.logger.info('Order enqueued', {'orderId': order['id']})
    return {'orderId': order['id']}

iii.register_function("orders::create", create_order)

async def main():
    await iii.connect()
    print("Producer ready")
    stop = asyncio.Event()
    loop = asyncio.get_running_loop()
    for sig in (signal.SIGINT, signal.SIGTERM):
        loop.add_signal_handler(sig, stop.set)
    await stop.wait()

asyncio.run(main())

# Then call from another function or worker
result = await iii.trigger('orders::create', {'items': ['item-1', 'item-2'], 'total': 99.99})
print('Created order:', result['orderId'])
```
</Tab>
<Tab value="Rust">
```rust title="queue_producer.rs"
use iii_sdk::{get_context, III};
use serde_json::json;
use tokio::signal;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let url = std::env::var("III_BRIDGE_URL").unwrap_or_else(|_| "ws://127.0.0.1:49134".to_string());
    let iii = III::new(&url);
    iii.connect().await?;

    let iii_clone = iii.clone();
    iii.register_function("orders::create", move |input| {
        let iii = iii_clone.clone();
        async move {
            let ctx = get_context();
            let order_id = uuid::Uuid::new_v4().to_string();

            iii.trigger_void("enqueue", json!({
                "topic": "order.created",
                "data": { "id": order_id, "items": input["items"], "total": input["total"] },
            }))?;

            ctx.logger.info("Order enqueued", Some(json!({ "orderId": order_id })));
            Ok(json!({ "orderId": order_id }))
        }
    });

    println!("Producer ready");
    signal::ctrl_c().await?;
    Ok(())
}

// Then call from another function or worker
let result = iii.trigger("orders::create", json!({ "items": ["item-1", "item-2"], "total": 99.99 })).await?;
println!("Created order: {}", result["orderId"]);
```

### 3. Register a consumer Function

Register a function and bind it to a `queue` trigger with the same topic. The engine invokes this function whenever a message arrives on that topic.

<Tabs groupId="language" persist items={["Node / TypeScript", "Python", "Rust"]}>
<Tab value="Node / TypeScript">
```typescript title="queue-consumer.ts"
import { init, getContext } from 'iii-sdk'

const iii = init(process.env.III_BRIDGE_URL ?? 'ws://localhost:49134')

iii.registerFunction({ id: 'orders::process-payment' }, async (data) => {
  const { logger } = getContext()
  logger.info('Processing payment for order', { orderId: data.id })
  // ...payment logic here...

  iii.triggerVoid('enqueue', {
    topic: 'order.paid',
    data: { orderId: data.id },
  })

  return { processed: true }
})

iii.registerTrigger({
  type: 'queue',
  function_id: 'orders::process-payment',
  config: { topic: 'order.created' },
})
```
</Tab>
<Tab value="Python">
```python title="queue_consumer.py"
import asyncio
import os
import signal
from iii import III
from iii.context import get_context

iii = III(os.environ.get('III_BRIDGE_URL', 'ws://localhost:49134'))

async def process_payment(data):
    ctx = get_context()
    ctx.logger.info('Processing payment for order', {'orderId': data['id']})
    print("Processing payment for order", data["id"])
    # ...payment logic here...

    iii.trigger_void("enqueue", {
        "topic": "order.paid",
        "data": {"orderId": data["id"]},
    })

    return {'processed': True}

iii.register_function("orders::process-payment", process_payment)
iii.register_trigger(
    type="queue",
    function_id="orders::process-payment",
    config={"topic": "order.created"},
)

async def main():
    await iii.connect()
    print("Consumer ready for order.created")
    stop = asyncio.Event()
    loop = asyncio.get_running_loop()
    for sig in (signal.SIGINT, signal.SIGTERM):
        loop.add_signal_handler(sig, stop.set)
    await stop.wait()

asyncio.run(main())
```
</Tab>
<Tab value="Rust">
```rust title="queue_consumer.rs"
use iii_sdk::{get_context, III};
use serde_json::json;
use tokio::signal;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let url = std::env::var("III_BRIDGE_URL").unwrap_or_else(|_| "ws://127.0.0.1:49134".to_string());
    let iii = III::new(&url);
    iii.connect().await?;

    let iii_clone = iii.clone();
    iii.register_function("orders::process-payment", move |data| {
        let iii = iii_clone.clone();
        async move {
            let ctx = get_context();
            let order_id = data["id"].as_str().unwrap_or("");
            ctx.logger.info("Processing payment for order", Some(json!({ "orderId": order_id })));
            println!("Processing payment for order {}", order_id);
            // ...payment logic here...

            iii.trigger_void("enqueue", json!({
                "topic": "order.paid",
                "data": { "orderId": order_id },
            }))?;

            Ok(json!({ "processed": true }))
        }
    });

    iii.register_trigger("queue", "orders::process-payment", json!({
        "topic": "order.created",
    }))?;

    println!("Consumer ready for order.created");
    signal::ctrl_c().await?;
    Ok(())
}
```

### 4. Chain multiple consumers (fan-out)

Multiple functions can subscribe to the same topic. When `order.created` fires, both `process-payment` and `send-confirmation` run independently:

<Tabs groupId="language" persist items={["Node / TypeScript", "Python", "Rust"]}>
<Tab value="Node / TypeScript">
```typescript title="queue-fanout.ts"
iii.registerFunction({ id: 'orders::send-confirmation' }, async (data) => {
  const { logger } = getContext()
  // ...send email logic here...
  logger.info('Confirmation sent for order', { orderId: data.id })
  return { sent: true }
})

iii.registerTrigger({
  type: 'queue',
  function_id: 'orders::send-confirmation',
  config: { topic: 'order.created' },
})
```
</Tab>

### 5. Conditional queue triggers

Use `condition_function_id` to only invoke the handler when a condition is met:

<Tabs groupId="language" persist items={["Node / TypeScript", "Python", "Rust"]}>
<Tab value="Node / TypeScript">
```typescript title="queue-conditional.ts"
iii.registerFunction({ id: 'orders::high-value-alert' }, async (data) => {
  const { logger } = getContext()
  // ...notify Slack logic here...
  logger.info('High-value order', { orderId: data.id, total: data.total })
  return { alerted: true }
})

const highValueCheck = iii.registerFunction({ id: 'orders::is-high-value' }, async (data) => {
  return data.total > 1000
})

iii.registerTrigger({
  type: 'queue',
  function_id: 'orders::high-value-alert',
  config: {
    topic: 'order.created',
    condition_function_id: highValueCheck.id,
  },
})
```
</Tab>
<Tab value="Python">
```python title="queue_conditional.py"
async def high_value_alert(data):
    ctx = get_context()
    # ...notify Slack logic here...
    ctx.logger.info('High-value order', {'orderId': data['id'], 'total': data['total']})
    print(f"High-value order: {data['id']} (${data['total']})")
    return {'alerted': True}

async def is_high_value(data):
    return data.get("total", 0) > 1000

iii.register_function("orders::high-value-alert", high_value_alert)
high_value_check = iii.register_function("orders::is-high-value", is_high_value)

iii.register_trigger(
    type="queue",
    function_id="orders::high-value-alert",
    config={
        "topic": "order.created",
        "condition_function_id": high_value_check.id,
    },
)
```
</Tab>
<Tab value="Rust">
```rust title="queue_conditional.rs"
iii.register_function("orders::high-value-alert", |data| async move {
    let ctx = get_context();
    let id = data["id"].as_str().unwrap_or("");
    let total = data["total"].as_f64().unwrap_or(0.0);
    // ...notify Slack logic here...
    ctx.logger.info("High-value order", Some(json!({ "orderId": id, "total": total })));
    println!("High-value order: {} (${})", id, total);
    Ok(json!({ "alerted": true }))
});

let high_value_check = iii.register_function("orders::is-high-value", |data| async move {
    let total = data["total"].as_f64().unwrap_or(0.0);
    Ok(json!(total > 1000.0))
});

iii.register_trigger("queue", "orders::high-value-alert", json!({
    "topic": "order.created",
    "condition_function_id": high_value_check.id,
}))?;
```

## Result

The producer returns immediately after enqueueing. The queue adapter distributes the message to all subscribed functions, which process it independently on their own workers. If a consumer fails, it does not affect the producer or other consumers.

## Queue vs triggerVoid

| | `enqueue` | `triggerVoid()` |
|---|---|---|
| **Decoupling** | Producer doesn't know consumers | Producer must trigger the consumer |
| **Fan-out** | Multiple consumers per topic | One target function |
| **Persistence** | Messages survive restarts (when stored) | No persistence |
| **Scaling** | Consumers scale independently | Producer must be updated for each new consumer |
