## Goal

Run a Function whenever a specific state value changes, using a `state` trigger.

## Steps

### 1. Enable the State module

```yaml title="iii-config.yaml"
modules:
  - class: modules::state::StateModule
    config:
      adapter:
        class: modules::state::adapters::KvStore
        config:
          store_method: file_based  # Options: in_memory, file_based
          # file_path: ./data/state_store.db  # required for file_based
```

### 2. Register the Function

The `state` trigger fires whenever the watched `scope`/`key` pair changes. The handler receives an event with `key`, `new_value`, `old_value`, `scope`, and `event_type` properties.

```typescript title="state-watcher.ts"
import { init, getContext } from 'iii-sdk'

const iii = init(process.env.III_BRIDGE_URL ?? 'ws://localhost:49134')

iii.registerFunction({ id: 'orders::on-status-change' }, async (event) => {
  const { logger } = getContext()
  logger.info('Order status changed', { key: event.key, new_value: event.new_value })
  // ...react to the change here...
  return { handled: true }
})

// Then call from another function or worker to trigger the state change
await iii.trigger('state::set', {
  scope: 'orders',
  key: 'status',
  value: { orderId: 'order-123', status: 'shipped' },
})
```

<Tab value="Python">
```python title="state_watcher.py"
import asyncio
import os
import signal
from iii import III
from iii.context import get_context

iii = III(os.environ.get('III_BRIDGE_URL', 'ws://localhost:49134'))

async def on_status_change(event):
    ctx = get_context()
    ctx.logger.info('Order status changed', {'key': event['key'], 'new_value': event['new_value']})
    print(f"Order status changed: key={event['key']}, new_value={event['new_value']}")
    # ...react to the change here...
    return {'handled': True}

iii.register_function('orders::on-status-change', on_status_change)

async def main():
    await iii.connect()
    print("Watching orders/status for changes")
    stop = asyncio.Event()
    loop = asyncio.get_running_loop()
    for sig in (signal.SIGINT, signal.SIGTERM):
        loop.add_signal_handler(sig, stop.set)
    await stop.wait()

asyncio.run(main())

# Then call from another function or worker to trigger the state change
await iii.trigger('state::set', {
    'scope': 'orders',
    'key': 'status',
    'value': {'orderId': 'order-123', 'status': 'shipped'},
})
```
</Tab>
<Tab value="Rust">
```rust title="state_watcher.rs"
use iii_sdk::{get_context, III};
use serde_json::json;
use tokio::signal;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let url = std::env::var("III_BRIDGE_URL").unwrap_or_else(|_| "ws://127.0.0.1:49134".to_string());
    let iii = III::new(&url);
    iii.connect().await?;

    iii.register_function("orders::on-status-change", |event| async move {
        let ctx = get_context();
        ctx.logger.info("Order status changed", Some(json!({ "key": event["key"], "new_value": event["new_value"] })));
        println!(
            "Order status changed: key={}, new_value={}",
            event["key"], event["new_value"]
        );
        // ...react to the change here...
        Ok(json!({ "handled": true }))
    });

    println!("Watching orders/status for changes");
    signal::ctrl_c().await?;
    Ok(())
}

// Then call from another function or worker to trigger the state change
iii.trigger("state::set", json!({
    "scope": "orders",
    "key": "status",
    "value": { "orderId": "order-123", "status": "shipped" },
})).await?;
```

### 4. Watch multiple keys

Each state trigger watches a single `scope`/`key` pair. To react to multiple keys, register a function and trigger for each:

<Tabs groupId="language" persist items={["Node / TypeScript", "Python", "Rust"]}>
<Tab value="Node / TypeScript">
```typescript title="multi-key-watcher.ts"
iii.registerFunction({ id: 'orders::on-total-change' }, async (event) => {
  const { logger } = getContext()
  logger.info('Order total changed', { new_value: event.new_value })
  return { handled: true }
})

iii.registerTrigger({
  type: 'state',
  function_id: 'orders::on-status-change',
  config: { scope: 'orders', key: 'status' },
})

iii.registerTrigger({
  type: 'state',
  function_id: 'orders::on-total-change',
  config: { scope: 'orders', key: 'total' },
})
```
</Tab>
<Tab value="Python">
```python title="multi_key_watcher.py"
async def on_total_change(event):
    ctx = get_context()

    print(f"Order total changed: {event['new_value']}")

iii.register_function('orders::on-total-change', on_total_change)

iii.register_trigger(
    type="state",
    function_id="orders::on-status-change",
    config={"scope": "orders", "key": "status"},
)

iii.register_trigger(
    type="state",
    function_id="orders::on-total-change",
    config={"scope": "orders", "key": "total"},
)
```
</Tab>
<Tab value="Rust">

```rust title="multi_key_watcher.rs"
iii.register_function("orders::on-total-change", |event| async move {

    println!("Order total changed: {}", event["new_value"]);

});

iii.register_trigger(
    "state",

```
</Tab>

## Result

When any Function writes to the watched `scope`/`key` pair via `state::set` or `state::update`, the Engine automatically invokes the registered handler with the new value. No polling needed.
